Example

$("#stepwizard").bhHeader({
    items: [
        { stepId: "step1", title: "步骤向导-1" },
        { stepId: "step2", title: "步骤向导-2" },
        { stepId: "step3", title: "步骤向导-3" }
    ],
    active: "step3",//可选, 当前激活项的stepId
    finished: ['step2'], //可选, 当前已完成项的stepId数组,默认值为[]
    change: function () { } //可选, 焦点项变化的回调,默认值为null
});
